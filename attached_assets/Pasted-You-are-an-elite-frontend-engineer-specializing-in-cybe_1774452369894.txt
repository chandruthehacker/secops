==================================================
GOAL
==================================================

Build the complete frontend for "SecOps Console" — a mini SIEM (Security Information and Event Management) web application with a detection engine UI. This is a SOC-style security operations dashboard that displays security logs from multiple sources, runs detection rules against them, generates alerts mapped to MITRE ATT&CK, and provides a full analyst investigation workflow.

There is NO backend. All data is mock data stored locally. All interactions use local state. The architecture must be cleanly separated so that mock data services can be swapped for real API calls later with minimal refactoring.

Do NOT skip any page. Do NOT simplify any feature. Do NOT use placeholder text where real mock data should exist. Every page must be fully functional with working search, filters, sorting, clicking, status changes, and navigation.

export type LogSourceType = 'windows_event' | 'linux_syslog' | 'linux_auth' | 'firewall' | 'auth' | 'apache_access';

export interface RawLog {
  id: string;
  source: LogSourceType;
  rawText: string;
  receivedAt: string; // ISO timestamp
}

export interface UnifiedLog {
  id: string;
  timestamp: string;             // ISO timestamp of the event
  source: LogSourceType;
  sourceIp: string;
  destinationIp?: string;
  sourcePort?: number;
  destinationPort?: number;
  hostname: string;
  username?: string;
  eventType: string;              // e.g., "authentication_failure", "process_creation", "connection_blocked"
  action: string;                 // e.g., "blocked", "allowed", "failed", "success"
  severity: 'info' | 'low' | 'medium' | 'high' | 'critical';
  message: string;
  rawLog: string;                 // Original raw log text
  parsedFields: Record<string, string | number | boolean>;
  tags: string[];
}

export interface LogSource {
  id: string;
  name: string;
  type: LogSourceType;
  status: 'active' | 'inactive' | 'error';

  description: string;

  description: string;
  severity: AlertSeverity;
  status: AlertStatus;
  source: LogSourceType;
  ruleId: string;

  updatedAt: string;              // ISO timestamp
  assignedTo?: string;            // Analyst name

export type RuleStatus = 'enabled' | 'disabled' | 'testing';
export type RuleSeverity = 'critical' | 'high' | 'medium' | 'low' | 'info';

  name: string;
  description: string;
  severity: RuleSeverity;
  status: RuleStatus;
  logSources: LogSourceType[];

    count: number;
    timeWindowMinutes: number;
    groupBy?: string;

  mitreTactic: string;
  mitreTechnique: string;
  mitreId: string;
  tags: string[];
  createdAt: string;
  updatedAt: string;
  author: string;
  falsePositiveRate?: number;     // 0-100
  alertCount: number;             // Total alerts generated
  yamlDefinition: string;         // Sigma-like YAML representation
}

  value: string | number | string[];
}

TypeScript

// services/alertService.ts
import { mockAlerts } from '../data/mockAlerts';
import { Alert, AlertStatus } from '../types/alert';

// TODO: Replace with actual API calls
const API_BASE = '/api'; // Ready for future backend

export const alertService = {
  async getAlerts(): Promise<Alert[]> {
    // Simulate API delay
    await new Promise(resolve => setTimeout(resolve, 300));
    return mockAlerts;
  },

  async getAlertById(id: string): Promise<Alert | undefined> {
    await new Promise(resolve => setTimeout(resolve, 200));
    return mockAlerts.find(a => a.id === id);
  },

  async updateAlertStatus(id: string, status: AlertStatus): Promise<Alert> {
    await new Promise(resolve => setTimeout(resolve, 200));
    const alert = mockAlerts.find(a => a.id === id);
    if (!alert) throw new Error('Alert not found');
    alert.status = status;
    return alert;
  },

  // ... more methods
};

This pattern is REQUIRED for ALL service files. Each function must be async and return a Promise, even though it's using local data. This makes future backend integration a simple swap.
